## Usage

```ts copy title="Import"
import {
  Masthead,
  MastheadHeader,
  MastheadContent,
  MastheadTitle,
  MastheadTrigger,
  MastheadSection,
} from "@govtechmy/myds-react/masthead";
```

```tsx copy title="Anatomy"
export default () => (
  <Masthead>
    <MastheadHeader>
      <MastheadTitle />
      <MastheadTrigger />
    </MastheadHeader>
    <MastheadContent>
      <MastheadSection />
    </MastheadContent>
  </Masthead>
);
```
