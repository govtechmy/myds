## Penggunaan

```ts copy title="Import"
import {
  Masthead,
  MastheadHeader,
  MastheadContent,
  MastheadTitle,
  MastheadTrigger,
  MastheadSection,
} from "@govtechmy/myds-react/masthead";
```

```tsx copy title="Anatomi"
export default () => (
  <Masthead>
    <MastheadHeader>
      <MastheadTitle />
      <MastheadTrigger />
    </MastheadHeader>
    <MastheadContent>
      <MastheadSection />
    </MastheadContent>
  </Masthead>
);
```
