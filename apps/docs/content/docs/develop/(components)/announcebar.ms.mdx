## Penggunaan

```ts copy title="Import"
import {
  AnnounceBar,
  AnnounceBarTag,
  AnnounceBarDescription,
} from "@govtechmy/myds-react/announce-bar";
```

```tsx copy title="Anatomi" title="Anatomi"
export default () => (
  <AnnounceBar>
    <AnnounceBarTag />
    <AnnounceBarDescription />
  </AnnounceBar>
);
```
