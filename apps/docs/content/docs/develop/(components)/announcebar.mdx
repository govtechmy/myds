## Usage

```tsx copy
import {
  AnnounceBar,
  AnnounceBarTag,
  AnnounceBarDescription,
} from "@govtechmy/myds-react/announce-bar";
```

```tsx copy
export default () => (
  <AnnounceBar>
    <AnnounceBarTag />
    <AnnounceBarDescription />
  </AnnounceBar>
);
```
